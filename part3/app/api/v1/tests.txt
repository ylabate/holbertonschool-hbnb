>>> from app import create_app, db
>>> from app.models.user import User
>>> app = create_app()
>>> ctx = app.app_context()
>>> ctx.push()
>>> db.drop_all()
>>> db.create_all()
>>> admin = User(first_name="Admin", last_name="Root", email="admin@example.com", password="adminpass", is_admin=True)
>>> owner = User(first_name="Alice", last_name="Owner", email="alice@example.com", password="ownerpass")
>>> reviewer = User(first_name="Bob", last_name="Reviewer", email="bob@example.com", password="reviewpass")
>>> db.session.add_all([admin, owner, reviewer])
>>> db.session.commit()
>>> client = app.test_client()

# Authentication
>>> admin_auth = client.post('/api/v1/auth/', json={"email": "admin@example.com", "password": "adminpass"})
>>> admin_auth.status_code
200
>>> admin_token = admin_auth.get_json()['access_token']

>>> owner_auth = client.post('/api/v1/auth/', json={"email": "alice@example.com", "password": "ownerpass"})
>>> owner_auth.status_code
200
>>> owner_token = owner_auth.get_json()['access_token']

>>> reviewer_auth = client.post('/api/v1/auth/', json={"email": "bob@example.com", "password": "reviewpass"})
>>> reviewer_auth.status_code
200
>>> reviewer_token = reviewer_auth.get_json()['access_token']

# Amenities
>>> response = client.post('/api/v1/amenities/', json={"name": "Wi-Fi", "description": "Fast internet"}, headers={"Authorization": f"Bearer {admin_token}"})
>>> response.status_code
201
>>> amenity1 = response.get_json()
>>> amenity1['name']
'Wi-Fi'

# Places
>>> response = client.post('/api/v1/places/', json={"title": "Cozy Apartment", "description": "Nice place", "price": 100.0, "latitude": 40.7128, "longitude": -74.0060, "amenities": [amenity1["id"]]}, headers={"Authorization": f"Bearer {owner_token}"})
>>> response.status_code
201
>>> place1 = response.get_json()
>>> place1['title']
'Cozy Apartment'

# Reviews
>>> response = client.post('/api/v1/reviews/', json={"text": "Great place!", "rating": 5, "user_id": reviewer.id, "place_id": place1["id"]}, headers={"Authorization": f"Bearer {reviewer_token}"})
>>> response.status_code
201
>>> review1 = response.get_json()
>>> review1['rating']
5

# Read endpoints
>>> client.get(f'/api/v1/amenities/{amenity1["id"]}').status_code
200
>>> client.get(f'/api/v1/places/{place1["id"]}').status_code
200
>>> client.get(f'/api/v1/reviews/{review1["id"]}').status_code
200

# Update endpoints
>>> response = client.put(f'/api/v1/amenities/{amenity1["id"]}', json={"name": "Fiber Wi-Fi", "description": "Updated"}, headers={"Authorization": f"Bearer {admin_token}"})
>>> response.status_code
200

>>> response = client.put(f'/api/v1/places/{place1["id"]}', json={"title": "Updated Apartment", "description": "Even nicer", "price": 150.0, "latitude": 40.7128, "longitude": -74.0060, "amenities": []}, headers={"Authorization": f"Bearer {owner_token}"})
>>> response.status_code
200

>>> response = client.put(f'/api/v1/reviews/{review1["id"]}', json={"text": "Updated review!", "rating": 4}, headers={"Authorization": f"Bearer {reviewer_token}"})
>>> response.status_code
200
>>> response.get_json()['rating']
4

# Delete review
>>> response = client.delete(f'/api/v1/reviews/{review1["id"]}', headers={"Authorization": f"Bearer {reviewer_token}"})
>>> response.status_code
200

>>> ctx.pop()
